FUNGSI EDIT DATA document.getElementById('btnEdit').onclick = function() { let terpilih = document.querySelectorAll('.cek-baris:checked'); if (terpilih.length === 0) { alert("Centang dulu kotak di sebelah kiri data, Pace!"); return; } terpilih.forEach(checkbox => { let baris = checkbox.closest('tr'); let selData = baris.querySelectorAll('.bisa-edit'); selData.forEach(td => { td.contentEditable = "true"; // Mengaktifkan mode ketik td.classList.add('sedang-diedit'); // Memberi warna kuning }); }); alert("Mode Edit Aktif! Silakan klik pada teks di tabel untuk merubahnya."); }; // 2. FUNGSI CARI (Berdasarkan Nama atau HP) function jalankanCari() { let input = document.getElementById('kotakCari').value.toLowerCase(); let barisTabel = document.querySelectorAll('#badanTabel tr'); barisTabel.forEach(tr => { let nama = tr.cells[1].innerText.toLowerCase(); let hp = tr.cells[4].innerText.toLowerCase(); if (nama.includes(input) || hp.includes(input)) { tr.style.display = ""; } else { tr.style.display = "none"; } }); } // 3. FUNGSI HAPUS document.getElementById('btnHapus').onclick = function() { let terpilih = document.querySelectorAll('.cek-baris:checked'); if (terpilih.length === 0) return alert("Pilih data yang mau dihapus!"); if (confirm("Hapus data terpilih?")) { terpilih.forEach(cb => cb.closest('tr').remove()); } }; // 4.
